API演示

使用Python快速测试我们的上下文管理API 

import requests

api_key = "demo_key_2026"
url = "https://api.ai-agent-hub.demo/context/query"

payload = {
    "agent_id": "demo_agent",
    "query": "最近的对话历史",
    "limit": 10
}

headers = {
    "Authorization": f"Bearer {api_key}",
    "Content-Type": "application/json"
}

response = requests.post(url, json=payload, headers=headers)
print(response.json())
